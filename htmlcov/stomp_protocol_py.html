98 """ 

99 Start a connection. 

100

102 :param str passcode: the password used to authenticate with 

103 :param bool wait: if True, wait for the connection to be established/acknowledged 

104 :param dict headers: a map of any additional headers the broker requires

500 def connect(self, username=None, passcode=None, wait=False, headers=None, 

501 with_connect_command=False, **keyword_headers): 

502 """ 

503 Send a STOMP CONNECT frame. Differs from 1.0 and 1.1 versions in that the HOST header is enforced. 

504 

505 :param str username: optionally specify the login user 

506 :param str passcode: optionally specify the user password 

507 :param bool wait: wait for the connection to complete before returning 

508 :param dict headers: a map of any additional headers to send with the subscription 

509 :param with_connect_command: if True, use CONNECT command instead of STOMP 

510 :param keyword_headers: any additional headers to send with the subscription 

511 """ 

512 cmd = CMD_CONNECT if with_connect_command else CMD_STOMP 

513 headers = utils.merge_headers([headers, keyword_headers]) 

514 headers[HDR_ACCEPT_VERSION] = self.version 

515 headers[HDR_HOST] = self.transport.current_host_and_port[0] 

516 

517 if self.transport.vhost: 

518 headers[HDR_HOST] = self.transport.vhost 

519 

520 if username is not None: 

521 headers[HDR_LOGIN] = username 

522 

523 if passcode is not None: 

524 headers[HDR_PASSCODE] = passcode 

525 

526 self.send_frame(cmd, headers) 

527 

528 if wait: 

529 self.transport.wait_for_connection() 

530 if self.transport.connection_error: 

531 raise ConnectFailedException()
