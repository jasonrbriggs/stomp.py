154class HeartbeatListener(ConnectionListener): 

155 """ 

156 Listener used to handle STOMP heartbeating. 

157 """ 

158 def __init__(self, transport, heartbeats, heart_beat_receive_scale=1.5): 

159 self.running = False 

160 self.transport = transport 

161 self.heartbeats = heartbeats 

162 self.received_heartbeat = None 

163 self.heartbeat_thread = None 

164 self.next_outbound_heartbeat = None 

165 self.heart_beat_receive_scale = heart_beat_receive_scale 

166 self.heartbeat_terminate_event = threading.Event() 

167 self.disconnecting = False 

168 

169 def on_connected(self, frame): 

170 """ 

171 Once the connection is established, and 'heart-beat' is found in the headers, we calculate the real 

172 heartbeat numbers (based on what the server sent and what was specified by the client) - if the heartbeats 

173 are not 0, we start up the heartbeat loop accordingly. 

174 

175 :param Frame frame: the stomp frame 

176 """ 

177 self.disconnecting = False 

178 if "heart-beat" in frame.headers: 

179 self.heartbeats = utils.calculate_heartbeats( 

180 frame.headers["heart-beat"].replace(" ", "").split(","), self.heartbeats) 

181 logging.debug("Heartbeats calculated %s", str(self.heartbeats)) 

182 if self.heartbeats != (0, 0): 

183 self.send_sleep = self.heartbeats[0] / 1000 

184 

185 # by default, receive gets an additional grace of 50% 

186 # set a different heart-beat-receive-scale when creating the connection to override that 

187 self.receive_sleep = (self.heartbeats[1] / 1000) * self.heart_beat_receive_scale 

188 

189 logging.debug("Set receive_sleep to %s, send_sleep to %s", self.receive_sleep, self.send_sleep) 

190 

191 # Give grace of receiving the first heartbeat 

192 self.received_heartbeat = monotonic() + self.receive_sleep 

193 

194 self.running = True 

195 if self.heartbeat_thread is None: 

196 self.heartbeat_thread = utils.default_create_thread( 

197 self.__heartbeat_loop) 

198 self.heartbeat_thread.name = "StompHeartbeat%s" % \ 

199 getattr(self.heartbeat_thread, "name", "Thread") 

200 

201 def on_disconnected(self): 

202 self.running = False 

203 self.heartbeat_terminate_event.set() 

204 

205 def on_disconnecting(self): 

206 self.disconnecting = True 

207 

208 def on_message(self, frame): 

209 """ 

210 Reset the last received time whenever a message is received. 

211 

212 :param Frame frame: the stomp frame 

213 """ 

214 # reset the heartbeat for any received message 

215 self.__update_heartbeat() 

216 

217 def on_receipt(self, *_): 

218 """ 

219 Reset the last received time whenever a receipt is received. 

220 """ 

221 self.__update_heartbeat() 

222 

223 def on_error(self, *_): 

224 """ 

225 Reset the last received time whenever an error is received. 

226 """ 

227 self.__update_heartbeat() 

228 

229 def on_heartbeat(self): 

230 """ 

231 Reset the last received time whenever a heartbeat message is received. 

232 """ 

233 self.__update_heartbeat() 

234 

235 def on_send(self, frame): 

236 """ 

237 Add the heartbeat header to the frame when connecting, and bump 

238 next outbound heartbeat timestamp. 

239 

240 :param Frame frame: the Frame object 

241 """ 

242 if frame.cmd == CMD_CONNECT or frame.cmd == CMD_STOMP: 

243 if self.heartbeats != (0, 0): 

244 frame.headers[HDR_HEARTBEAT] = "%s,%s" % self.heartbeats 

245 if self.next_outbound_heartbeat is not None: 

246 self.next_outbound_heartbeat = monotonic() + self.send_sleep 

247 

248 def __update_heartbeat(self): 

249 # Honour any grace that has been already included 

250 if self.received_heartbeat is None: 

251 return 

252 now = monotonic() 

253 if now > self.received_heartbeat: 

254 self.received_heartbeat = now 

255 

256 def __heartbeat_loop(self): 

257 """ 

258 Main loop for sending (and monitoring received) heartbeats. 

259 """ 

260 logging.info("Starting heartbeat loop") 

261 now = monotonic() 

262 

263 # Setup the initial due time for the outbound heartbeat 

264 if self.send_sleep != 0: 

265 self.next_outbound_heartbeat = now + self.send_sleep 

266 logging.debug("Calculated next outbound heartbeat as %s", self.next_outbound_heartbeat) 

267 

268 while self.running: 

269 now = monotonic() 

270 

271 next_events = [] 

272 if self.next_outbound_heartbeat is not None: 

273 next_events.append(self.next_outbound_heartbeat - now) 

274 if self.receive_sleep != 0: 

275 t = self.received_heartbeat + self.receive_sleep - now 

276 if t > 0: 

277 next_events.append(t) 

278 sleep_time = min(next_events) 

279 if sleep_time > 0: 

280 terminate = self.heartbeat_terminate_event.wait(sleep_time) 

281 if terminate: 

282 break 

283 

284 now = monotonic() 

285 

286 if not self.transport.is_connected() or self.disconnecting: 

287 time.sleep(self.send_sleep) 

288 continue 

289 

290 if self.send_sleep != 0 and now > self.next_outbound_heartbeat: 

291 logging.debug("Sending a heartbeat message at %s", now) 

292 try: 

293 self.transport.transmit(utils.Frame(None, {}, None)) 

294 except exception.NotConnectedException: 

295 logging.debug("Lost connection, unable to send heartbeat") 

296 except Exception: 

297 _, e, _ = sys.exc_info() 

298 logging.debug("Unable to send heartbeat, due to: %s", e) 

299 

300 if self.receive_sleep != 0: 

301 diff_receive = now - self.received_heartbeat 

302 

303 if diff_receive > self.receive_sleep: 

304 # heartbeat timeout 

305 logging.warning("Heartbeat timeout: diff_receive=%s, time=%s, lastrec=%s", 

306 diff_receive, now, self.received_heartbeat) 

307 self.transport.set_connected(False) 

308 self.transport.disconnect_socket() 

309 self.transport.stop() 

310 for listener in self.transport.listeners.values(): 

311 listener.on_heartbeat_timeout() 

312 self.heartbeat_thread = None 

313 self.heartbeat_terminate_event.clear() 

314 if self.heartbeats != (0, 0): 

315 # don't bother logging this if heartbeats weren't setup to start with 

316 logging.info("Heartbeat loop ended")

319class WaitingListener(ConnectionListener): 

320 """ 

321 A listener which waits for a specific receipt to arrive. 

322 """ 

323 def __init__(self, receipt): 

324 """ 

325 :param str receipt: 

326 """ 

327 self.receipt_condition = threading.Condition() 

328 self.disconnect_condition = threading.Condition() 

329 self.receipt = receipt 

330 self.received = False 

331 self.disconnected = False 

332 

333 def on_receipt(self, frame): 

334 """ 

335 If the receipt id can be found in the headers, then notify the waiting thread. 

336 

337 :param Frame frame: the stomp frame 

338 """ 

339 if "receipt-id" in frame.headers and frame.headers["receipt-id"] == self.receipt: 

340 with self.receipt_condition: 

341 self.received = True 

342 self.receipt_condition.notify() 

343 

344 def on_disconnected(self): 

345 with self.disconnect_condition: 

346 self.disconnected = True 

347 self.disconnect_condition.notify() 

348 

349 def wait_on_receipt(self): 

350 """ 

351 Wait until we receive a message receipt. 

352 """ 

353 with self.receipt_condition: 

354 while not self.received: 

355 self.receipt_condition.wait() 

356 self.received = False 

357 

358 def wait_on_disconnected(self): 

359 """ 

360 Wait until disconnected. 

361 """ 

362 with self.disconnect_condition: 

363 while not self.disconnected: 

364 self.disconnect_condition.wait()
