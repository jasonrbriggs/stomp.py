96# Used to parse STOMP header lines in the format "key:value", 

97# 

98HEADER_LINE_RE = re.compile("(?P<key>[^:]+)[:](?P<value>.*)") 

99 

100## 

101# As of STOMP 1.2, lines can end with either line feed, or carriage return plus line feed. 

102# 

103PREAMBLE_END_RE = re.compile(b"\n\n|\r\n\r\n") 

104 

105## 

106# As of STOMP 1.2, lines can end with either line feed, or carriage return plus line feed. 

107# 

108LINE_END_RE = re.compile("\n|\r\n") 

109 

110## 

111# Used to replace the "passcode" to be dumped in the transport log (at debug level)

113PASSCODE_RE = re.compile(r"passcode:\s+[^\' ]+")

149 "\r": "\\r", 

150 "\n": "\\n", 

151 ":": "\\c",

157def _unescape_header(matchobj): 

158 escaped = matchobj.group(0) 

159 unescaped = _HEADER_UNESCAPES.get(escaped) 

160 if not unescaped: 

161 # TODO: unknown escapes MUST be treated as fatal protocol error per spec 

162 unescaped = escaped 

163 return unescaped

166def parse_headers(lines, offset=0): 

167 """ 

168 Parse the headers in a STOMP response 

169 

170 :param list(str) lines: the lines received in the message response 

171 :param int offset: the starting line number 

172 

173 :rtype: dict(str,str) 

174 """ 

175 headers = {} 

176 for header_line in lines[offset:]: 

177 header_match = HEADER_LINE_RE.match(header_line) 

178 if header_match: 

179 key = header_match.group("key") 

180 key = re.sub(r"\\.", _unescape_header, key) 

181 if key not in headers: 

182 value = header_match.group("value") 

183 value = re.sub(r"\\.", _unescape_header, value) 

184 headers[key] = value 

185 return headers

188def parse_frame(frame): 

189 """ 

190 Parse a STOMP frame into a Frame object. 

191 

192 :param bytes frame: the frame received from the server (as a byte string) 

193 

194 :rtype: Frame 

195 """ 

196 mat = PREAMBLE_END_RE.search(frame) 

197 if mat: 

198 preamble_end = mat.start() 

199 body_start = mat.end() 

200 else: 

201 preamble_end = len(frame) 

202 body_start = preamble_end 

203 preamble = decode(frame[0:preamble_end]) 

204 preamble_lines = LINE_END_RE.split(preamble) 

205 preamble_len = len(preamble_lines) 

206 body = frame[body_start:] 

207 

208 # Skip any leading newlines 

209 first_line = 0 

210 while first_line < preamble_len and len(preamble_lines[first_line]) == 0: 

211 first_line += 1 

212 

213 if first_line >= preamble_len: 

214 return None 

215 

216 # Extract frame type/command 

217 cmd = preamble_lines[first_line] 

218 

219 # Put headers into a key/value map 

220 headers = parse_headers(preamble_lines, first_line + 1) 

221 

222 return Frame(cmd, headers, body)

253def calculate_heartbeats(shb, chb): 

254 """ 

255 Given a heartbeat string from the server, and a heartbeat tuple from the client, 

256 calculate what the actual heartbeat settings should be. 

257 

258 :param (str,str) shb: server heartbeat numbers 

259 :param (int,int) chb: client heartbeat numbers 

260 

261 :rtype: (int,int) 

262 """ 

263 (sx, sy) = shb 

264 (cx, cy) = chb 

265 x = 0 

266 y = 0 

267 if cx != 0 and sy != "0": 

268 x = max(cx, int(sy)) 

269 if cy != 0 and sx != "0": 

270 y = max(cy, int(sx)) 

271 return x, y

274def convert_frame(frame): 

275 """ 

276 Convert a frame to a list of lines separated by newlines. 

277 

278 :param Frame frame: the Frame object to convert 

279 

280 :rtype: list(str) 

281 """ 

282 lines = [] 

283 

284 body = None 

285 if frame.body: 

286 body = encode(frame.body) 

287 

288 if HDR_CONTENT_LENGTH in frame.headers: 

289 frame.headers[HDR_CONTENT_LENGTH] = len(body) 

290 

291 if frame.cmd: 

292 lines.append(encode(frame.cmd)) 

293 lines.append(ENC_NEWLINE) 

294 for key, vals in sorted(frame.headers.items()): 

295 if vals is None: 

296 continue 

297 if type(vals) != tuple: 

298 vals = (vals,) 

299 for val in vals: 

300 lines.append(encode("%s:%s\n" % (key, val))) 

301 lines.append(ENC_NEWLINE) 

302 if body: 

303 lines.append(body) 

304 

305 if frame.cmd: 

306 lines.append(ENC_NULL) 

307 return lines

323class Frame(object): 

324 """ 

325 A STOMP frame (or message). 

326 

327 :param str cmd: the protocol command 

328 :param dict headers: a map of headers for the frame 

329 :param body: the content of the frame. 

330 """ 

331 def __init__(self, cmd, headers=None, body=None): 

332 self.cmd = cmd 

333 self.headers = headers if headers is not None else {} 

334 self.original_headers = copy.copy(self.headers) 

335 self.body = body 

336 

337 def __str__(self): 

338 return "{cmd=%s,headers=[%s],body=%s}" % (self.cmd, self.headers, self.body)
