94# Used to parse STOMP header lines in the format "key:value", 

95# 

96HEADER_LINE_RE = re.compile("(?P<key>[^:]+)[:](?P<value>.*)") 

97 

98## 

99# As of STOMP 1.2, lines can end with either line feed, or carriage return plus line feed. 

100# 

101PREAMBLE_END_RE = re.compile(b"\n\n|\r\n\r\n")

103## 

104# As of STOMP 1.2, lines can end with either line feed, or carriage return plus line feed. 

105# 

106LINE_END_RE = re.compile("\n|\r\n")

109# Used to replace the "passcode" to be dumped in the transport log (at debug level)

147 '\r': '\\r', 

148 '\n': '\\n', 

149 ':': '\\c',

155def _unescape_header(matchobj): 

156 escaped = matchobj.group(0) 

157 unescaped = _HEADER_UNESCAPES.get(escaped) 

158 if not unescaped: 

159 # TODO: unknown escapes MUST be treated as fatal protocol error per spec 

160 unescaped = escaped 

161 return unescaped

164def parse_headers(lines, offset=0): 

165 """ 

166 Parse the headers in a STOMP response 

167 

168 :param list(str) lines: the lines received in the message response 

169 :param int offset: the starting line number 

170 

171 :rtype: dict(str,str) 

172 """ 

173 headers = {} 

174 for header_line in lines[offset:]: 

175 header_match = HEADER_LINE_RE.match(header_line) 

176 if header_match: 

177 key = header_match.group("key") 

178 key = re.sub(r'\\.', _unescape_header, key) 

179 if key not in headers: 

180 value = header_match.group("value") 

181 value = re.sub(r'\\.', _unescape_header, value) 

182 headers[key] = value 

183 return headers

186def parse_frame(frame): 

187 """ 

188 Parse a STOMP frame into a Frame object. 

189 

190 :param bytes frame: the frame received from the server (as a byte string) 

191 

192 :rtype: Frame 

193 """ 

194 mat = PREAMBLE_END_RE.search(frame) 

195 if mat: 

196 preamble_end = mat.start() 

197 body_start = mat.end() 

198 else: 

199 preamble_end = len(frame) 

200 body_start = preamble_end 

201 preamble = decode(frame[0:preamble_end]) 

202 preamble_lines = LINE_END_RE.split(preamble) 

203 preamble_len = len(preamble_lines) 

204 body = frame[body_start:] 

205 

206 # Skip any leading newlines 

207 first_line = 0 

208 while first_line < preamble_len and len(preamble_lines[first_line]) == 0: 

209 first_line += 1 

210 

211 if first_line >= preamble_len: 

212 return None 

213 

214 # Extract frame type/command 

215 cmd = preamble_lines[first_line] 

216 

217 # Put headers into a key/value map 

218 headers = parse_headers(preamble_lines, first_line + 1) 

219 

220 return Frame(cmd, headers, body)

251def calculate_heartbeats(shb, chb): 

252 """ 

253 Given a heartbeat string from the server, and a heartbeat tuple from the client, 

254 calculate what the actual heartbeat settings should be. 

255 

256 :param (str,str) shb: server heartbeat numbers 

257 :param (int,int) chb: client heartbeat numbers 

258 

259 :rtype: (int,int) 

260 """ 

261 (sx, sy) = shb 

262 (cx, cy) = chb 

263 x = 0 

264 y = 0 

265 if cx != 0 and sy != '0': 

266 x = max(cx, int(sy)) 

267 if cy != 0 and sx != '0': 

268 y = max(cy, int(sx)) 

269 return x, y

272def convert_frame(frame): 

273 """ 

274 Convert a frame to a list of lines separated by newlines. 

275 

276 :param Frame frame: the Frame object to convert 

277 

278 :rtype: list(str) 

279 """ 

280 lines = [] 

281 

282 body = None 

283 if frame.body: 

284 body = encode(frame.body) 

285 

286 if HDR_CONTENT_LENGTH in frame.headers: 

287 frame.headers[HDR_CONTENT_LENGTH] = len(body) 

288 

289 if frame.cmd: 

290 lines.append(encode(frame.cmd)) 

291 lines.append(ENC_NEWLINE) 

292 for key, vals in sorted(frame.headers.items()): 

293 if vals is None: 

294 continue 

295 if type(vals) != tuple: 

296 vals = (vals,) 

297 for val in vals: 

298 lines.append(encode("%s:%s\n" % (key, val))) 

299 lines.append(ENC_NEWLINE) 

300 if body: 

301 lines.append(body) 

302 

303 if frame.cmd: 

304 lines.append(ENC_NULL) 

305 return lines

321class Frame(object): 

322 """ 

323 A STOMP frame (or message). 

324 

325 :param str cmd: the protocol command 

326 :param dict headers: a map of headers for the frame 

327 :param body: the content of the frame. 

328 """ 

329 def __init__(self, cmd, headers=None, body=None): 

330 self.cmd = cmd 

331 self.headers = headers if headers is not None else {} 

332 self.original_headers = copy.copy(self.headers) 

333 self.body = body 

334 

335 def __str__(self): 

336 return "{cmd=%s,headers=[%s],body=%s}" % (self.cmd, self.headers, self.body)
