49def check_ssl_certificate(host_and_port): 

50 ''' 

51 Check the expiry date of the certificate presented by the host/port. 

52 ''' 

53 if OpenSSL: 

54 host, port = host_and_port 

55 cert = ssl.get_server_certificate((host, port)) 

56 x509 = OpenSSL.crypto.load_certificate(OpenSSL.crypto.FILETYPE_PEM, cert) 

57 dt = datetime.datetime.strptime(x509.get_notAfter().decode().replace('Z',''), "%Y%m%d%H%M%S") 

58 valid_cert = dt > datetime.datetime.now() 

59 if not valid_cert: 

60 logging.info("SSL certificate for %s:%s expired on %s", host, port, dt) 

61 assert valid_cert

201 def process_frame(self, f, frame_str): 

202 """ 

203 :param Frame f: Frame object 

204 :param bytes frame_str: raw frame content 

205 """ 

206 frame_type = f.cmd.lower() 

207 if frame_type in ["connected", "message", "receipt", "error", "heartbeat"]: 

208 if frame_type == "message": 

209 self.notify("before_message", f)

218 def notify(self, frame_type, frame=None): 

219 """ 

220 Utility function for notifying listeners of incoming and outgoing messages 

221 

222 :param str frame_type: the type of message 

223 :param Frame frame: the stomp frame 

224 """ 

225 if frame_type == "receipt": 

226 # logic for wait-on-receipt notification 

227 receipt = frame.headers["receipt-id"] 

228 receipt_value = self.__receipts.get(receipt) 

229 with self.__send_wait_condition: 

230 self.set_receipt(receipt, None) 

231 self.__send_wait_condition.notify() 

232 

233 if receipt_value == CMD_DISCONNECT: 

234 self.set_connected(False) 

235 # received a stomp 1.1+ disconnect receipt 

236 if receipt == self.__disconnect_receipt: 

237 self.disconnect_socket() 

238 self.__disconnect_receipt = None 

239 

240 elif frame_type == "connected": 

241 self.set_connected(True) 

242 

243 elif frame_type == "disconnected": 

244 self.notified_on_disconnect = True 

245 self.set_connected(False) 

246 

247 with self.__listeners_change_condition: 

248 listeners = sorted(self.listeners.items()) 

249 

250 for (_, listener) in listeners: 

251 notify_func = getattr(listener, "on_%s" % frame_type, None) 

252 if not notify_func: 

253 logging.debug("listener %s has no method on_%s", listener, frame_type) 

254 continue 

255 if frame_type in ("heartbeat", "disconnected"): 

256 notify_func() 

257 continue 

258 if frame_type == "connecting": 

259 notify_func(self.current_host_and_port) 

260 continue 

261 

262 if frame_type == "error" and not self.connected: 

263 with self.__connect_wait_condition: 

264 self.connection_error = True 

265 self.__connect_wait_condition.notify() 

266 

267 notify_func(frame)

346 def __receiver_loop(self): 

347 """ 

348 Main loop listening for incoming data. 

349 """ 

350 logging.info("Starting receiver loop (%s)", threading.current_thread()) 

351 notify_disconnected = True 

352 try: 

353 while self.running: 

354 try: 

355 while self.running: 

356 frames = self.__read() 

357 

358 for frame in frames: 

359 if self.__is_eol(frame): 

360 f = HEARTBEAT_FRAME 

361 else: 

362 f = parse_frame(frame) 

363 if f is None: 

364 continue 

365 if self.__auto_decode: 

366 f.body = decode(f.body) 

367 self.process_frame(f, frame) 

368 except exception.ConnectionClosedException: 

369 if self.running: 

370 # 

371 # Clear out any half-received messages after losing connection 

372 # 

373 self.__recvbuf = b"" 

374 self.running = False 

375 notify_disconnected = True 

376 break 

377 finally: 

378 self.cleanup() 

379 finally: 

380 with self.__receiver_thread_exit_condition: 

381 self.__receiver_thread_exited = True 

382 self.__receiver_thread_exit_condition.notifyAll() 

383 logging.info("Receiver loop ended") 

384 self.notify("receiver_loop_completed") 

385 if notify_disconnected and not self.notified_on_disconnect: 

386 self.notify("disconnected") 

387 with self.__connect_wait_condition: 

388 self.__connect_wait_condition.notifyAll() 

389 self.notified_on_disconnect = False

391 def __read(self): 

392 """ 

393 Read the next frame(s) from the socket. 

394 

395 :return: list of frames read 

396 :rtype: list(bytes) 

397 """ 

398 fastbuf = BytesIO() 

399 while self.running: 

400 try: 

401 try: 

402 c = self.receive() 

403 except exception.InterruptedException: 

404 logging.debug("socket read interrupted, restarting") 

405 continue 

406 except Exception: 

407 logging.debug("socket read error", exc_info=logging.verbose) 

408 c = b"" 

409 if c is None or len(c) == 0: 

410 logging.debug("nothing received, raising CCE") 

411 raise exception.ConnectionClosedException() 

412 if self.__is_eol(c) and not self.__recvbuf and not fastbuf.tell(): 

413 # 

414 # EOL to an empty receive buffer: treat as heartbeat. 

415 # Note that this may misdetect an optional EOL at end of frame as heartbeat in case the 

416 # previous receive() got a complete frame whose NUL at end of frame happened to be the 

417 # last byte of that read. But that should be harmless in practice. 

418 # 

419 fastbuf.close() 

420 return [c] 

421 fastbuf.write(c) 

422 if b"\x00" in c: 

423 # 

424 # Possible end of frame 

425 # 

426 break 

427 self.__recvbuf += fastbuf.getvalue() 

428 fastbuf.close() 

429 result = [] 

430 

431 if self.__recvbuf and self.running: 

432 while True: 

433 pos = self.__recvbuf.find(b"\x00") 

434 

435 if pos >= 0: 

436 frame = self.__recvbuf[0:pos] 

437 preamble_end_match = PREAMBLE_END_RE.search(frame) 

438 if preamble_end_match: 

439 preamble_end = preamble_end_match.start() 

440 content_length_match = BaseTransport.__content_length_re.search(frame[0:preamble_end]) 

441 if content_length_match: 

442 content_length = int(content_length_match.group("value")) 

443 content_offset = preamble_end_match.end() 

444 frame_size = content_offset + content_length 

445 if frame_size > len(frame): 

446 # 

447 # Frame contains NUL bytes, need to read more 

448 # 

449 if frame_size < len(self.__recvbuf): 

450 pos = frame_size 

451 frame = self.__recvbuf[0:pos] 

452 else: 

453 # 

454 # Haven't read enough data yet, exit loop and wait for more to arrive 

455 # 

456 break 

457 result.append(frame) 

458 pos += 1 

459 # 

460 # Ignore optional EOLs at end of frame 

461 # 

462 while 1: 

463 if self.__is_eol(self.__recvbuf[pos:pos + 1]): 

464 pos += 1 

465 elif self.__is_eol(self.__recvbuf[pos:pos + 2]): 

466 pos += 2 

467 else: 

468 break 

469 self.__recvbuf = self.__recvbuf[pos:] 

470 else: 

471 break 

472 return result

475class Transport(BaseTransport): 

476 """ 

477 Represents a STOMP client 'transport'. Effectively this is the communications mechanism without the definition of 

478 the protocol. 

479 

480 :param list((str,int)) host_and_ports: a list of (host, port) tuples 

481 :param bool prefer_localhost: if True and the local host is mentioned in the (host, 

482 port) tuples, try to connect to this first 

483 :param bool try_loopback_connect: if True and the local host is found in the host 

484 tuples, try connecting to it using loopback interface 

485 (127.0.0.1) 

486 :param float reconnect_sleep_initial: initial delay in seconds to wait before reattempting 

487 to establish a connection if connection to any of the 

488 hosts fails. 

489 :param float reconnect_sleep_increase: factor by which the sleep delay is increased after 

490 each connection attempt. For example, 0.5 means 

491 to wait 50% longer than before the previous attempt, 

492 1.0 means wait twice as long, and 0.0 means keep 

493 the delay constant. 

494 :param float reconnect_sleep_max: maximum delay between connection attempts, regardless 

495 of the reconnect_sleep_increase. 

496 :param float reconnect_sleep_jitter: random additional time to wait (as a percentage of 

497 the time determined using the previous parameters) 

498 between connection attempts in order to avoid 

499 stampeding. For example, a value of 0.1 means to wait 

500 an extra 0%-10% (randomly determined) of the delay 

501 calculated using the previous three parameters. 

502 :param int reconnect_attempts_max: maximum attempts to reconnect (Can also be used for infinite attempts : `-1`) 

503 :param timeout: the timeout value to use when connecting the stomp socket 

504 :param keepalive: some operating systems support sending the occasional heart 

505 beat packets to detect when a connection fails. This 

506 parameter can either be set set to a boolean to turn on the 

507 default keepalive options for your OS, or as a tuple of 

508 values, which also enables keepalive packets, but specifies 

509 options specific to your OS implementation. 

510 For linux, supply ("linux", ka_idle, ka_intvl, ka_cnt) 

511 For macos, supply ("mac", ka_intvl) 

512 :param str vhost: specify a virtual hostname to provide in the 'host' header of the connection 

513 :param int recv_bytes: the number of bytes to use when calling recv 

514 """

599 def disconnect_socket(self): 

600 """ 

601 Disconnect the underlying socket connection 

602 """ 

603 self.running = False 

604 if self.socket is not None: 

605 if self.__need_ssl(): 

606 # 

607 # Even though we don't want to use the socket, unwrap is the only API method which does a proper SSL 

608 # shutdown 

609 # 

610 try: 

611 self.socket = self.socket.unwrap() 

612 except Exception: 

613 # 

614 # unwrap seems flaky on Win with the back-ported ssl mod, so catch any exception and log it 

615 # 

616 _, e, _ = sys.exc_info() 

617 logging.warning(e) 

618 elif hasattr(socket, "SHUT_RDWR"): 

619 try: 

620 self.socket.shutdown(socket.SHUT_RDWR) 

621 except socket.error: 

622 _, e, _ = sys.exc_info() 

623 # ignore when socket already closed 

624 if get_errno(e) != errno.ENOTCONN: 

625 logging.warning("Unable to issue SHUT_RDWR on socket because of error '%s'", e) 

626 

627 # 

628 # split this into a separate check, because sometimes the socket is nulled between shutdown and this call 

629 # 

630 if self.socket is not None: 

631 try: 

632 self.socket.close() 

633 except socket.error: 

634 _, e, _ = sys.exc_info() 

635 logging.warning("Unable to close socket because of error '%s'", e) 

636 self.current_host_and_port = None 

637 self.socket = None 

638 if not self.notified_on_disconnect: 

639 self.notify("disconnected")

708 if ka_sig == "auto": 

709 if LINUX_KEEPALIVE_AVAIL: 

710 ka_sig = "linux" 

711 ka_args = None 

712 logging.info("keepalive: autodetected linux-style support") 

713 elif MAC_KEEPALIVE_AVAIL: 

714 ka_sig = "mac" 

715 ka_args = None 

716 logging.info("keepalive: autodetected mac-style support") 

717 else: 

718 logging.error("keepalive: unable to detect any implementation, DISABLED!") 

719 return 

720 

721 if ka_sig == "linux": 

722 logging.info("keepalive: activating linux-style support") 

723 if ka_args is None: 

724 logging.info("keepalive: using system defaults") 

725 ka_args = (None, None, None) 

726 ka_idle, ka_intvl, ka_cnt = ka_args 

727 if try_setsockopt(self.socket, "enable", SOL_SOCKET, SO_KEEPALIVE, 1): 

728 try_setsockopt(self.socket, "idle time", SOL_TCP, TCP_KEEPIDLE, ka_idle) 

729 try_setsockopt(self.socket, "interval", SOL_TCP, TCP_KEEPINTVL, ka_intvl) 

730 try_setsockopt(self.socket, "count", SOL_TCP, TCP_KEEPCNT, ka_cnt) 

731 elif ka_sig == "mac": 

732 logging.info("keepalive: activating mac-style support") 

733 if ka_args is None: 

734 logging.info("keepalive: using system defaults") 

735 ka_args = (3,) 

736 ka_intvl = ka_args 

737 if try_setsockopt(self.socket, "enable", SOL_SOCKET, SO_KEEPALIVE, 1): 

738 try_setsockopt(self.socket, socket.IPPROTO_TCP, 0x10, ka_intvl) 

739 else: 

740 logging.error("keepalive: implementation %r not recognized or not supported", ka_sig)

742 def attempt_connection(self): 

743 """ 

744 Try connecting to the (host, port) tuples specified at construction time. 

745 """ 

746 self.connection_error = False 

747 sleep_exp = 1 

748 connect_count = 0 

749 

750 logging.info("attempt reconnection (%s, %s, %s)", self.running, self.socket, connect_count) 

751 while self.running and self.socket is None and (connect_count < self.__reconnect_attempts_max or 

752 self.__reconnect_attempts_max == -1): 

753 for host_and_port in self.__host_and_ports: 

754 try: 

755 logging.info("Attempting connection to host %s, port %s", host_and_port[0], host_and_port[1]) 

756 if self.__bind_host_port: 

757 self.socket = socket.create_connection(host_and_port, self.__timeout, self.__bind_host_port) 

758 else: 

759 self.socket = socket.create_connection(host_and_port, self.__timeout) 

760 self.__enable_keepalive() 

761 need_ssl = self.__need_ssl(host_and_port) 

762 

763 if need_ssl: # wrap socket 

764 check_ssl_certificate(host_and_port) 

765 ssl_params = self.get_ssl(host_and_port) 

766 tls_context = ssl.SSLContext(DEFAULT_SSL_VERSION) 

767 if ssl_params["ca_certs"]: 

768 cert_validation = ssl.CERT_REQUIRED 

769 tls_context.load_verify_locations(ssl_params["ca_certs"]) 

770 else: 

771 cert_validation = ssl.CERT_NONE 

772 if tls_context: 

773 # Wrap the socket for TLS 

774 certfile = ssl_params["cert_file"] 

775 keyfile = ssl_params["key_file"] 

776 password = ssl_params.get("password") 

777 if certfile and not keyfile: 

778 keyfile = certfile 

779 if certfile: 

780 tls_context.load_cert_chain(certfile, keyfile, password) 

781 if cert_validation is None or cert_validation == ssl.CERT_NONE: 

782 tls_context.check_hostname = False 

783 tls_context.verify_mode = cert_validation 

784 logging.info("Wrapping SSL socket") 

785 self.socket = tls_context.wrap_socket(self.socket, server_hostname=host_and_port[0]) 

786 else: 

787 # Old-style wrap_socket where we don't have a modern SSLContext (so no SNI) 

788 logging.info("Wrapping SSL socket (old style)") 

789 self.socket = ssl.wrap_socket( 

790 self.socket, 

791 keyfile=ssl_params["key_file"], 

792 certfile=ssl_params["cert_file"], 

793 cert_reqs=cert_validation, 

794 ca_certs=ssl_params["ca_certs"], 

795 ssl_version=ssl_params["ssl_version"]) 

796 

797 self.socket.settimeout(self.__timeout) 

798 

799 if self.blocking is not None: 

800 self.socket.setblocking(self.blocking) 

801 

802 # 

803 # Validate server cert 

804 # 

805 if need_ssl and ssl_params["cert_validator"]: 

806 cert = self.socket.getpeercert() 

807 (ok, errmsg) = ssl_params["cert_validator"](cert, host_and_port[0]) 

808 if not ok: 

809 raise SSLError("Server certificate validation failed: %s", errmsg) 

810 

811 self.current_host_and_port = host_and_port 

812 logging.info("Established connection to host %s, port %s", host_and_port[0], host_and_port[1]) 

813 break 

814 except (OSError, AssertionError): 

815 self.socket = None 

816 connect_count += 1 

817 logging.warning("Could not connect to host %s, port %s", host_and_port[0], host_and_port[1], 

818 exc_info=logging.verbose) 

819 

820 if self.socket is None: 

821 sleep_duration = (min(self.__reconnect_sleep_max, 

822 ((self.__reconnect_sleep_initial / (1.0 + self.__reconnect_sleep_increase)) 

823 * math.pow(1.0 + self.__reconnect_sleep_increase, sleep_exp))) 

824 * (1.0 + random.random() * self.__reconnect_sleep_jitter)) 

825 sleep_end = monotonic() + sleep_duration 

826 logging.debug("Sleeping for %.1f seconds before attempting reconnect", sleep_duration) 

827 while self.running and monotonic() < sleep_end: 

828 time.sleep(0.2) 

829 

830 if sleep_duration < self.__reconnect_sleep_max: 

831 sleep_exp += 1 

832 

833 if not self.socket: 

834 raise exception.ConnectFailedException()

836 def set_ssl(self, 

837 for_hosts=[], 

838 key_file=None, 

839 cert_file=None, 

840 ca_certs=None, 

841 cert_validator=None, 

842 ssl_version=DEFAULT_SSL_VERSION, 

843 password=None): 

844 """ 

845 Sets up SSL configuration for the given hosts. This ensures socket is wrapped in a SSL connection, raising an 

846 exception if the SSL module can't be found. 

847 

848 :param for_hosts: a list of tuples describing hosts this SSL configuration should be applied to 

849 :param cert_file: the path to a X509 certificate 

850 :param key_file: the path to a X509 key file 

851 :param ca_certs: the path to the a file containing CA certificates to validate the server against. 

852 If this is not set, server side certificate validation is not done. 

853 :param cert_validator: function which performs extra validation on the client certificate, for example 

854 checking the returned certificate has a commonName attribute equal to the 

855 hostname (to avoid man in the middle attacks). 

856 The signature is: (OK, err_msg) = validation_function(cert, hostname) 

857 where OK is a boolean, and cert is a certificate structure 

858 as returned by ssl.SSLSocket.getpeercert() 

859 :param ssl_version: SSL protocol to use for the connection. This should be one of the PROTOCOL_x 

860 constants provided by the ssl module. The default is ssl.PROTOCOL_TLSv1 

861 :param password: SSL password 

862 """ 

863 if not ssl: 

864 raise Exception("SSL connection requested, but SSL library not found") 

865 

866 for host_port in for_hosts: 

867 self.__ssl_params[host_port] = dict(key_file=key_file, 

868 cert_file=cert_file, 

869 ca_certs=ca_certs, 

870 cert_validator=cert_validator, 

871 ssl_version=ssl_version, 

872 password=password)
