105 def __print_async(self, frame_type, frame): 

106 """ 

107 Utility function to print a message and setup the command prompt 

108 for the next input 

109 """ 

110 if self.__quit: 

111 return 

112 if self.verbose: 

113 self.__sysout(frame_type) 

114 for k, v in frame.headers.items(): 

115 self.__sysout("%s: %s" % (k, v)) 

116 else: 

117 if "message-id" in frame.headers: 

118 self.__sysout("message-id: %s" % frame.headers["message-id"]) 

119 if "subscription" in frame.headers: 

120 self.__sysout("subscription: %s" % frame.headers["subscription"]) 

121 if self.prompt != "": 

122 self.__sysout("") 

123 self.__sysout(frame.body) 

124 if not self.__start: 

125 self.__sysout(self.prompt, end="") 

126 else: 

127 self.__start = False 

128 self.stdout.flush()

243 def do_subscribe(self, args): 

244 args = args.split() 

245 if len(args) < 1: 

246 self.__error("Expecting: subscribe <destination> [ack]") 

247 return 

248 

249 name = args[0] 

250 if name in self.__subscriptions: 

251 self.__error("Already subscribed to %s" % name) 

252 return 

253 

254 ack_mode = "auto" 

255 if len(args) >= 2: 

256 ack_mode = args[1] 

257 

258 sid = self.__subscription_id 

259 self.__subscription_id += 1 

260 

261 self.__sysout("Subscribing to '%s' with acknowledge set to '%s', id set to '%s'" % (name, ack_mode, sid)) 

262 self.conn.subscribe(destination=name, ack=ack_mode, id=sid) 

263 self.__subscriptions[name] = SubscriptionInfo(sid, ack_mode)

380 def check_ack_nack(self, acknackfunc, args): 

381 if self.nversion >= 1.2 and len(args) < 1: 

382 self.__error("Expecting: %s <ack-id>" % acknackfunc) 

383 return None 

384 elif self.nversion == 1.1 and len(args) < 2: 

385 self.__error("Expecting: %s <message-id> <subscription-id>" % acknackfunc) 

386 return None 

387 elif len(args) < 1: 

388 self.__error("Expecting: %s <message-id>" % acknackfunc) 

389 return None 

390 

391 if self.nversion == 1.1: 

392 return partial(acknackfunc, args[0], args[1]) 

393 else: 

394 return partial(acknackfunc, args[0])

506def main(): 

507 arguments = docopt(__doc__, version=version_string)

518 heartbeats = tuple(map(int, arguments["--heartbeats"].split(","))) 

519 

520 st = StompCLI(arguments["--host"], arguments["--port"], arguments["--user"], arguments["--password"], arguments["--protocol"],

522 use_ssl=arguments["--ssl"], 

523 ssl_key_file=arguments["--ssl-key-file"], 

524 ssl_cert_file=arguments["--ssl-cert-file"], 

525 ssl_ca_file=arguments["--ssl-ca-file"])

527 if arguments["--listen"] is not None: 

528 st.do_subscribe(arguments["--listen"]) 

529 try:

534 elif arguments["--file"] is not None:
