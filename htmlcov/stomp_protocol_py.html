96 def connect(self, username=None, passcode=None, wait=False, headers=None, **keyword_headers): 

97 """ 

98 Start a connection. 

99 

100 :param str username: the username to connect with 

101 :param str passcode: the password used to authenticate with 

102 :param bool wait: if True, wait for the connection to be established/acknowledged 

103 :param dict headers: a map of any additional headers the broker requires 

104 :param keyword_headers: any additional headers the broker requires 

105 """ 

106 cmd = CMD_CONNECT 

107 headers = utils.merge_headers([headers, keyword_headers]) 

108 headers[HDR_ACCEPT_VERSION] = self.version 

109 

110 if username is not None: 

111 headers[HDR_LOGIN] = username

113 if passcode is not None: 

114 headers[HDR_PASSCODE] = passcode 

115 

116 self.send_frame(cmd, headers) 

117 

118 if wait: 

119 self.transport.wait_for_connection() 

120 if self.transport.connection_error: 

121 raise ConnectFailedException()

487 """ 

488 Send a STOMP CONNECT frame. Differs from 1.0 and 1.1 versions in that the HOST header is enforced. 

489 

490 :param str username: optionally specify the login user 

491 :param str passcode: optionally specify the user password 

492 :param bool wait: wait for the connection to complete before returning 

493 :param dict headers: a map of any additional headers to send with the subscription 

494 :param keyword_headers: any additional headers to send with the subscription 

495 """ 

496 cmd = CMD_STOMP 

497 headers = utils.merge_headers([headers, keyword_headers]) 

498 headers[HDR_ACCEPT_VERSION] = self.version 

499 headers[HDR_HOST] = self.transport.current_host_and_port[0] 

500 

501 if self.transport.vhost: 

502 headers[HDR_HOST] = self.transport.vhost 

503 

504 if username is not None: 

505 headers[HDR_LOGIN] = username 

506 

507 if passcode is not None: 

508 headers[HDR_PASSCODE] = passcode 

509 

510 self.send_frame(cmd, headers) 

511 

512 if wait: 

513 self.transport.wait_for_connection() 

514 if self.transport.connection_error: 

515 raise ConnectFailedException()
