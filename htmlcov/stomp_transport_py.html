174 def process_frame(self, f, frame_str): 

175 """ 

176 :param Frame f: Frame object 

177 :param bytes frame_str: raw frame content 

178 """ 

179 frame_type = f.cmd.lower() 

180 if frame_type in ['connected', 'message', 'receipt', 'error', 'heartbeat']: 

181 if frame_type == 'message': 

182 (f.headers, f.body) = self.notify('before_message', f.headers, f.body) 

183 if logging.isEnabledFor(logging.DEBUG): 

184 logging.debug("Received frame: %r, headers=%r, body=%r", f.cmd, f.headers, f.body) 

185 else: 

186 logging.info("Received frame: %r, len(body)=%r", f.cmd, length(f.body)) 

187 self.notify(frame_type, f.headers, f.body) 

188 else: 

189 logging.warning("Unknown response frame type: '%s' (frame length was %d)", frame_type, length(frame_str))

191 def notify(self, frame_type, headers=None, body=None): 

192 """ 

193 Utility function for notifying listeners of incoming and outgoing messages 

194 

195 :param str frame_type: the type of message 

196 :param dict headers: the map of headers associated with the message 

197 :param body: the content of the message 

198 """ 

199 if frame_type == 'receipt': 

200 # logic for wait-on-receipt notification 

201 receipt = headers['receipt-id'] 

202 receipt_value = self.__receipts.get(receipt) 

203 with self.__send_wait_condition: 

204 self.set_receipt(receipt, None) 

205 self.__send_wait_condition.notify() 

206 

207 if receipt_value == CMD_DISCONNECT: 

208 self.set_connected(False) 

209 # received a stomp 1.1+ disconnect receipt 

210 if receipt == self.__disconnect_receipt: 

211 self.disconnect_socket() 

212 self.__disconnect_receipt = None 

213 

214 elif frame_type == 'connected': 

215 self.set_connected(True) 

216 

217 elif frame_type == 'disconnected': 

218 self.__notified_on_disconnect = True 

219 self.set_connected(False) 

220 

221 with self.__listeners_change_condition: 

222 listeners = sorted(self.listeners.items()) 

223 

224 for (_, listener) in listeners: 

225 notify_func = getattr(listener, 'on_%s' % frame_type, None) 

226 if not notify_func: 

227 logging.debug("listener %s has no method on_%s", listener, frame_type) 

228 continue 

229 if frame_type in ('heartbeat', 'disconnected'): 

230 notify_func() 

231 continue 

232 if frame_type == 'connecting': 

233 notify_func(self.current_host_and_port) 

234 continue 

235 

236 if frame_type == 'error' and not self.connected: 

237 with self.__connect_wait_condition: 

238 self.connection_error = True 

239 self.__connect_wait_condition.notify() 

240 

241 rtn = notify_func(headers, body) 

242 if rtn: 

243 (headers, body) = rtn 

244 return (headers, body)

323 def __receiver_loop(self): 

324 """ 

325 Main loop listening for incoming data. 

326 """ 

327 logging.info("Starting receiver loop (%s)" % threading.current_thread()) 

328 notify_disconnected = True 

329 try: 

330 while self.running: 

331 try: 

332 while self.running: 

333 frames = self.__read() 

334 

335 for frame in frames: 

336 f = parse_frame(frame) 

337 if f is None: 

338 continue 

339 if self.__auto_decode: 

340 f.body = decode(f.body) 

341 self.process_frame(f, frame) 

342 except exception.ConnectionClosedException: 

343 if self.running: 

344 # 

345 # Clear out any half-received messages after losing connection 

346 # 

347 self.__recvbuf = b'' 

348 self.running = False 

349 notify_disconnected = True 

350 break 

351 finally: 

352 self.cleanup() 

353 finally: 

354 with self.__receiver_thread_exit_condition: 

355 self.__receiver_thread_exited = True 

356 self.__receiver_thread_exit_condition.notifyAll() 

357 logging.info("Receiver loop ended") 

358 self.notify("receiver_loop_completed") 

359 if notify_disconnected and not self.__notified_on_disconnect: 

360 self.notify('disconnected') 

361 with self.__connect_wait_condition: 

362 self.__connect_wait_condition.notifyAll()

364 def __read(self): 

365 """ 

366 Read the next frame(s) from the socket. 

367 

368 :return: list of frames read 

369 :rtype: list(bytes) 

370 """ 

371 fastbuf = BytesIO() 

372 while self.running: 

373 try: 

374 try: 

375 c = self.receive() 

376 except exception.InterruptedException: 

377 logging.debug("socket read interrupted, restarting") 

378 continue 

379 except Exception: 

380 logging.debug("socket read error", exc_info=True) 

381 c = b'' 

382 if c is None or len(c) == 0: 

383 logging.debug("nothing received, raising CCE") 

384 raise exception.ConnectionClosedException() 

385 if c == b'\x0a' and not self.__recvbuf and not fastbuf.tell(): 

386 # 

387 # EOL to an empty receive buffer: treat as heartbeat. 

388 # Note that this may misdetect an optional EOL at end of frame as heartbeat in case the 

389 # previous receive() got a complete frame whose NUL at end of frame happened to be the 

390 # last byte of that read. But that should be harmless in practice. 

391 # 

392 fastbuf.close() 

393 return [c] 

394 fastbuf.write(c) 

395 if b'\x00' in c: 

396 # 

397 # Possible end of frame 

398 # 

399 break 

400 self.__recvbuf += fastbuf.getvalue() 

401 fastbuf.close() 

402 result = [] 

403 

404 if self.__recvbuf and self.running: 

405 while True: 

406 pos = self.__recvbuf.find(b'\x00') 

407 

408 if pos >= 0: 

409 frame = self.__recvbuf[0:pos] 

410 preamble_end_match = PREAMBLE_END_RE.search(frame) 

411 if preamble_end_match: 

412 preamble_end = preamble_end_match.start() 

413 content_length_match = BaseTransport.__content_length_re.search(frame[0:preamble_end]) 

414 if content_length_match: 

415 content_length = int(content_length_match.group('value')) 

416 content_offset = preamble_end_match.end() 

417 frame_size = content_offset + content_length 

418 if frame_size > len(frame): 

419 # 

420 # Frame contains NUL bytes, need to read more 

421 # 

422 if frame_size < len(self.__recvbuf): 

423 pos = frame_size 

424 frame = self.__recvbuf[0:pos] 

425 else: 

426 # 

427 # Haven't read enough data yet, exit loop and wait for more to arrive 

428 # 

429 break 

430 result.append(frame) 

431 pos += 1 

432 # 

433 # Ignore optional EOLs at end of frame 

434 # 

435 while self.__recvbuf[pos:pos + 1] == b'\x0a': 

436 pos += 1 

437 self.__recvbuf = self.__recvbuf[pos:] 

438 else: 

439 break 

440 return result

443class Transport(BaseTransport): 

444 """ 

445 Represents a STOMP client 'transport'. Effectively this is the communications mechanism without the definition of 

446 the protocol. 

447 

448 :param list((str,int)) host_and_ports: a list of (host, port) tuples 

449 :param bool prefer_localhost: if True and the local host is mentioned in the (host, 

450 port) tuples, try to connect to this first 

451 :param bool try_loopback_connect: if True and the local host is found in the host 

452 tuples, try connecting to it using loopback interface 

453 (127.0.0.1) 

454 :param float reconnect_sleep_initial: initial delay in seconds to wait before reattempting 

455 to establish a connection if connection to any of the 

456 hosts fails. 

457 :param float reconnect_sleep_increase: factor by which the sleep delay is increased after 

458 each connection attempt. For example, 0.5 means 

459 to wait 50% longer than before the previous attempt, 

460 1.0 means wait twice as long, and 0.0 means keep 

461 the delay constant. 

462 :param float reconnect_sleep_max: maximum delay between connection attempts, regardless 

463 of the reconnect_sleep_increase. 

464 :param float reconnect_sleep_jitter: random additional time to wait (as a percentage of 

465 the time determined using the previous parameters) 

466 between connection attempts in order to avoid 

467 stampeding. For example, a value of 0.1 means to wait 

468 an extra 0%-10% (randomly determined) of the delay 

469 calculated using the previous three parameters. 

470 :param int reconnect_attempts_max: maximum attempts to reconnect (Can also be used for infinite attempts : `-1`) 

471 :param bool use_ssl: deprecated, see :py:meth:`set_ssl` 

472 :param ssl_cert_file: deprecated, see :py:meth:`set_ssl` 

473 :param ssl_key_file: deprecated, see :py:meth:`set_ssl` 

474 :param ssl_ca_certs: deprecated, see :py:meth:`set_ssl` 

475 :param ssl_cert_validator: deprecated, see :py:meth:`set_ssl` 

476 :param ssl_version: deprecated, see :py:meth:`set_ssl` 

477 :param timeout: the timeout value to use when connecting the stomp socket 

478 :param keepalive: some operating systems support sending the occasional heart 

479 beat packets to detect when a connection fails. This 

480 parameter can either be set set to a boolean to turn on the 

481 default keepalive options for your OS, or as a tuple of 

482 values, which also enables keepalive packets, but specifies 

483 options specific to your OS implementation 

484 :param str vhost: specify a virtual hostname to provide in the 'host' header of the connection 

485 :param int recv_bytes: the number of bytes to use when calling recv 

486 """

582 def disconnect_socket(self): 

583 """ 

584 Disconnect the underlying socket connection 

585 """ 

586 self.running = False 

587 if self.socket is not None: 

588 if self.__need_ssl(): 

589 # 

590 # Even though we don't want to use the socket, unwrap is the only API method which does a proper SSL 

591 # shutdown 

592 # 

593 try: 

594 self.socket = self.socket.unwrap() 

595 except Exception: 

596 # 

597 # unwrap seems flaky on Win with the back-ported ssl mod, so catch any exception and log it 

598 # 

599 _, e, _ = sys.exc_info() 

600 logging.warning(e) 

601 elif hasattr(socket, 'SHUT_RDWR'): 

602 try: 

603 self.socket.shutdown(socket.SHUT_RDWR) 

604 except socket.error: 

605 _, e, _ = sys.exc_info() 

606 # ignore when socket already closed 

607 if get_errno(e) != errno.ENOTCONN: 

608 logging.warning("Unable to issue SHUT_RDWR on socket because of error '%s'", e) 

609 

610 # 

611 # split this into a separate check, because sometimes the socket is nulled between shutdown and this call 

612 # 

613 if self.socket is not None: 

614 try: 

615 self.socket.close() 

616 except socket.error: 

617 _, e, _ = sys.exc_info() 

618 logging.warning("Unable to close socket because of error '%s'", e) 

619 self.current_host_and_port = None 

620 self.socket = None

704 lka_idle, lka_intvl, lka_cnt = ka_args 

705 if try_setsockopt(self.socket, 'enable', SOL_SOCKET, SO_KEEPALIVE, 1): 

706 try_setsockopt(self.socket, 'idle time', SOL_TCP, TCP_KEEPIDLE, lka_idle) 

707 try_setsockopt(self.socket, 'interval', SOL_TCP, TCP_KEEPINTVL, lka_intvl) 

708 try_setsockopt(self.socket, 'count', SOL_TCP, TCP_KEEPCNT, lka_cnt) 

709 else: 

710 logging.error("keepalive: implementation %r not recognized or not supported", ka_sig)

712 def attempt_connection(self): 

713 """ 

714 Try connecting to the (host, port) tuples specified at construction time. 

715 """ 

716 self.connection_error = False 

717 sleep_exp = 1 

718 connect_count = 0 

719 

720 logging.info("attempt reconnection (%s, %s, %s)" % (self.running, self.socket, connect_count)) 

721 while self.running and self.socket is None and ( 

722 connect_count < self.__reconnect_attempts_max or 

723 self.__reconnect_attempts_max == -1 ): 

724 for host_and_port in self.__host_and_ports: 

725 try: 

726 logging.info("Attempting connection to host %s, port %s", host_and_port[0], host_and_port[1]) 

727 self.socket = socket.create_connection(host_and_port, self.__timeout) 

728 self.__enable_keepalive() 

729 need_ssl = self.__need_ssl(host_and_port) 

730 

731 if need_ssl: # wrap socket 

732 ssl_params = self.get_ssl(host_and_port) 

733 if ssl_params['ca_certs']: 

734 cert_validation = ssl.CERT_REQUIRED 

735 else: 

736 cert_validation = ssl.CERT_NONE 

737 try: 

738 tls_context = ssl.create_default_context(cafile=ssl_params['ca_certs']) 

739 except AttributeError: 

740 tls_context = None 

741 if tls_context: 

742 # Wrap the socket for TLS 

743 certfile = ssl_params['cert_file'] 

744 keyfile = ssl_params['key_file'] 

745 password = ssl_params.get('password') 

746 if certfile and not keyfile: 

747 keyfile = certfile 

748 if certfile: 

749 tls_context.load_cert_chain(certfile, keyfile, password) 

750 if cert_validation is None or cert_validation == ssl.CERT_NONE: 

751 tls_context.check_hostname = False 

752 tls_context.verify_mode = cert_validation 

753 self.socket = tls_context.wrap_socket(self.socket, server_hostname=host_and_port[0]) 

754 else: 

755 # Old-style wrap_socket where we don't have a modern SSLContext (so no SNI) 

756 self.socket = ssl.wrap_socket( 

757 self.socket, 

758 keyfile=ssl_params['key_file'], 

759 certfile=ssl_params['cert_file'], 

760 cert_reqs=cert_validation, 

761 ca_certs=ssl_params['ca_certs'], 

762 ssl_version=ssl_params['ssl_version']) 

763 

764 self.socket.settimeout(self.__timeout) 

765 

766 if self.blocking is not None: 

767 self.socket.setblocking(self.blocking) 

768 

769 # 

770 # Validate server cert 

771 # 

772 if need_ssl and ssl_params['cert_validator']: 

773 cert = self.socket.getpeercert() 

774 (ok, errmsg) = ssl_params['cert_validator'](cert, host_and_port[0]) 

775 if not ok: 

776 raise SSLError("Server certificate validation failed: %s", errmsg) 

777 

778 self.current_host_and_port = host_and_port 

779 logging.info("Established connection to host %s, port %s", host_and_port[0], host_and_port[1]) 

780 break 

781 except socket.error: 

782 self.socket = None 

783 connect_count += 1 

784 logging.warning("Could not connect to host %s, port %s", host_and_port[0], host_and_port[1], exc_info=1) 

785 

786 if self.socket is None: 

787 sleep_duration = (min(self.__reconnect_sleep_max, 

788 ((self.__reconnect_sleep_initial / (1.0 + self.__reconnect_sleep_increase)) 

789 * math.pow(1.0 + self.__reconnect_sleep_increase, sleep_exp))) 

790 * (1.0 + random.random() * self.__reconnect_sleep_jitter)) 

791 sleep_end = monotonic() + sleep_duration 

792 logging.debug("Sleeping for %.1f seconds before attempting reconnect", sleep_duration) 

793 while self.running and monotonic() < sleep_end: 

794 time.sleep(0.2) 

795 

796 if sleep_duration < self.__reconnect_sleep_max: 

797 sleep_exp += 1 

798 

799 if not self.socket: 

800 raise exception.ConnectFailedException()

802 def set_ssl(self, 

803 for_hosts=[], 

804 key_file=None, 

805 cert_file=None, 

806 ca_certs=None, 

807 cert_validator=None, 

808 ssl_version=DEFAULT_SSL_VERSION, 

809 password=None): 

810 """ 

811 Sets up SSL configuration for the given hosts. This ensures socket is wrapped in a SSL connection, raising an 

812 exception if the SSL module can't be found. 

813 

814 :param for_hosts: a list of tuples describing hosts this SSL configuration should be applied to 

815 :param cert_file: the path to a X509 certificate 

816 :param key_file: the path to a X509 key file 

817 :param ca_certs: the path to the a file containing CA certificates to validate the server against. 

818 If this is not set, server side certificate validation is not done. 

819 :param cert_validator: function which performs extra validation on the client certificate, for example 

820 checking the returned certificate has a commonName attribute equal to the 

821 hostname (to avoid man in the middle attacks). 

822 The signature is: (OK, err_msg) = validation_function(cert, hostname) 

823 where OK is a boolean, and cert is a certificate structure 

824 as returned by ssl.SSLSocket.getpeercert() 

825 :param ssl_version: SSL protocol to use for the connection. This should be one of the PROTOCOL_x 

826 constants provided by the ssl module. The default is ssl.PROTOCOL_TLSv1 

827 """ 

828 if not ssl: 

829 raise Exception("SSL connection requested, but SSL library not found") 

830 

831 for host_port in for_hosts: 

832 self.__ssl_params[host_port] = dict(key_file=key_file, 

833 cert_file=cert_file, 

834 ca_certs=ca_certs, 

835 cert_validator=cert_validator, 

836 ssl_version=ssl_version, 

837 password=password)
