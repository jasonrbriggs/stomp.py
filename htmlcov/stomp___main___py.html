104 def __print_async(self, frame_type, headers, body): 

105 """ 

106 Utility function to print a message and setup the command prompt 

107 for the next input 

108 """ 

109 if self.__quit: 

110 return 

111 if self.verbose: 

112 self.__sysout(frame_type) 

113 for k, v in headers.items(): 

114 self.__sysout('%s: %s' % (k, v)) 

115 else: 

116 if 'message-id' in headers: 

117 self.__sysout('message-id: %s' % headers['message-id']) 

118 if 'subscription' in headers: 

119 self.__sysout('subscription: %s' % headers['subscription']) 

120 if self.prompt != '': 

121 self.__sysout('') 

122 self.__sysout(body) 

123 if not self.__start: 

124 self.__sysout(self.prompt, end='') 

125 else: 

126 self.__start = False 

127 self.stdout.flush()

243 def do_subscribe(self, args): 

244 args = args.split() 

245 if len(args) < 1: 

246 self.__error('Expecting: subscribe <destination> [ack]') 

247 return 

248 

249 name = args[0] 

250 if name in self.__subscriptions: 

251 self.__error('Already subscribed to %s' % name) 

252 return 

253 

254 ack_mode = 'auto' 

255 if len(args) >= 2: 

256 ack_mode = args[1] 

257 

258 sid = self.__subscription_id 

259 self.__subscription_id += 1 

260 

261 self.__sysout('Subscribing to "%s" with acknowledge set to "%s", id set to "%s"' % (name, ack_mode, sid)) 

262 self.conn.subscribe(destination=name, ack=ack_mode, id=sid) 

263 self.__subscriptions[name] = SubscriptionInfo(sid, ack_mode)
