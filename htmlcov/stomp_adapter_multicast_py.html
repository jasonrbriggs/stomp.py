27 def attempt_connection(self): 

28 """ 

29 Establish a multicast connection - uses 2 sockets (one for sending, the other for receiving) 

30 """ 

31 self.socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM, socket.IPPROTO_UDP) 

32 self.socket.setsockopt(socket.IPPROTO_IP, socket.IP_MULTICAST_TTL, 2) 

33 

34 self.receiver_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM, socket.IPPROTO_UDP) 

35 self.receiver_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) 

36 self.receiver_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEPORT, 1) 

37 self.receiver_socket.bind(("", MCAST_PORT)) 

38 mreq = struct.pack("4sl", socket.inet_aton(MCAST_GRP), socket.INADDR_ANY) 

39 self.receiver_socket.setsockopt(socket.IPPROTO_IP, socket.IP_ADD_MEMBERSHIP, mreq) 

40 

41 if not self.socket or not self.receiver_socket: 

42 raise exception.ConnectFailedException()

60 def process_frame(self, f, frame_str): 

61 """ 

62 :param Frame f: Frame object 

63 :param bytes frame_str: Raw frame content 

64 """ 

65 frame_type = f.cmd.lower() 

66 

67 if frame_type in ["disconnect"]: 

68 return 

69 

70 if frame_type == "send": 

71 frame_type = "message" 

72 f.cmd = "MESSAGE" 

73 

74 if frame_type in ["connected", "message", "receipt", "error", "heartbeat"]: 

75 if frame_type == "message": 

76 if f.headers["destination"] not in self.subscriptions.values(): 

77 return 

78 self.notify("before_message", f) 

79 self.notify(frame_type, f) 

80 if "receipt" in f.headers: 

81 receipt_frame = Frame("RECEIPT", {"receipt-id": f.headers["receipt"]}) 

82 lines = convert_frame(receipt_frame) 

83 self.send(encode(pack(lines))) 

84 logging.debug("Received frame: %r, headers=%r, body=%r", f.cmd, f.headers, f.body)
